  18.  18 // convert an array to a date.
  19.  19 // the array should mirror the parameters below
  20.  20 // note: all values past the year are optional and will default to the lowest possible value.
  21.  21 // [year, month, day , hour, minute, second, millisecond]
  22.  22 export function configFromArray (config) {
  23.  23     var i, date, input = [], currentDate, yearToUse;
  24.  24 
  25.  25     if (config._d) {
  26.  26         return;

  33.  33         dayOfYearFromWeekInfo(config);

  39.  39

  42.  42         }

  45.  45         config._a[MONTH] = date.getUTCMonth();
  46.  46         config._a[DATE] = date.getUTCDate();
  47.  47     }
  48.  48 
  49.  49     // Default to current date.
  50.  50     // * if no year, month, day of month are given, default to today
  51.  51     // * if day of month is given, default month and year
  52.  52     // * if month is given, default only year
  53.  53     // * if year is given, don't default anything

  55.  55         config._a[i] = input[i] = currentDate[i];
  56.  56     }
  57.  57 
  58.  58     // Zero out whatever was not defaulted, including time
  59.  59     for (; i < 7; i++) {

  61.  61     }
  62.  62 
  63.  63     // Check for 24:00:00.000
  64.  64     if (config._a[HOUR] === 24 &&
  65.  65             config._a[MINUTE] === 0 &&
  66.  66             config._a[SECOND] === 0 &&
  67.  67             config._a[MILLISECOND] === 0) {
  68.  68         config._nextDay = true;
  69.  69         config._a[HOUR] = 0;
  70.  70     }
  71.  71 
  72.  72     config._d = (config._useUTC ? createUTCDate : createDate).apply(null, input);
  73.  73     // Apply timezone offset from input. The actual utcOffset can be changed
  74.  74     // with parseZone.
  75.  75     if (config._tzm != null) {
  76.  76         config._d.setUTCMinutes(config._d.getUTCMinutes() - config._tzm);
  77.  77     }
  78.  78 
  79.  79     if (config._nextDay) {
  80.  80         config._a[HOUR] = 24;
  81.  81     }
  82.  82 }

  84.  84 function dayOfYearFromWeekInfo(config) {

  86.  86

  89.  89         dow = 1;
  90.  90         doy = 4;
  91.  91

  93.  93         // how we interpret now (local, utc, fixed offset). So create

  105. 105 
  106. 106         if (w.d != null) {

  113. 113             // local weekday -- counting starts from begining of week
  114. 114             weekday = w.e + dow;
  115. 115         } else {
  116. 116             // default to begining of week
  117. 117             weekday = dow;
  118. 118         }
  119. 119     }
  120. 120     temp = dayOfYearFromWeeks(weekYear, week, weekday, doy, dow);
  121. 121 
  122. 122     config._a[YEAR] = temp.year;
  123. 123     config._dayOfYear = temp.dayOfYear;
