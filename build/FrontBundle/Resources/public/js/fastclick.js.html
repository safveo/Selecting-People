  463. 463  return true;
  464. 464  }
  465. 465 
  466. 466  return false;
  467. 467  };
  468. 468 
  469. 469 
  470. 470  /**
  471. 471   * Update the last position.
  472. 472   *
  473. 473   * @param {Event} event
  474. 474   * @returns {boolean}
  475. 475   */
  476. 476  FastClick.prototype.onTouchMove = function(event) {
  477. 477  if (!this.trackingClick) {
  478. 478  return true;
  479. 479  }
  480. 480 
  481. 481  // If the touch has moved, cancel the click tracking
  482. 482  if (this.targetElement !== this.getTargetElementFromEventTarget(event.target) || this.touchHasMoved(event)) {
  483. 483  this.trackingClick = false;
  484. 484  this.targetElement = null;
  485. 485  }
  486. 486 
  487. 487  return true;
  488. 488  };

  491. 491  /**

  498. 498 
  499. 499  // Fast path for newer browsers supporting the HTML5 control attribute
  500. 500  if (labelElement.control !== undefined) {
  501. 501  return labelElement.control;
  502. 502  }
  503. 503 
  504. 504  // All browsers under test that support touch events also support the HTML5 htmlFor attribute
  505. 505  if (labelElement.htmlFor) {
  506. 506  return document.getElementById(labelElement.htmlFor);

  515. 515  /**

  517. 517   *
  518. 518   * @param {Event} event
  519. 519   * @returns {boolean}
  520. 520   */
  521. 521  FastClick.prototype.onTouchEnd = function(event) {
  522. 522  var forElement, trackingClickStart, targetTagName, scrollParent, touch, targetElement = this.targetElement;
  523. 523 
  524. 524  if (!this.trackingClick) {
  525. 525  return true;
  526. 526  }
  527. 527 
  528. 528  // Prevent phantom clicks on fast double-tap (issue #36)
  529. 529  if ((event.timeStamp - this.lastClickTime) < this.tapDelay) {
  530. 530  this.cancelNextClick = true;
  531. 531  return true;
  532. 532  }
  533. 533 
  534. 534  if ((event.timeStamp - this.trackingClickStart) > this.tapTimeout) {
  535. 535  return true;
  536. 536  }
  537. 537 
  538. 538  // Reset to prevent wrong click cancel on input (issue #156).
  539. 539  this.cancelNextClick = false;
  540. 540 
  541. 541  this.lastClickTime = event.timeStamp;
  542. 542 
  543. 543  trackingClickStart = this.trackingClickStart;
  544. 544  this.trackingClick = false;
  545. 545  this.trackingClickStart = 0;
  546. 546 
  547. 547  // On some iOS devices, the targetElement supplied with the event is invalid if the layer
  548. 548  // is performing a transition or scroll, and has to be re-detected manually. Note that
  549. 549  // for this to function correctly, it must be called *after* the event target is checked!
  550. 550  // See issue #57; also filed as rdar://13048589 .
  551. 551  if (deviceIsIOSWithBadTarget) {
  552. 552  touch = event.changedTouches[0];
  553. 553 
  554. 554  // In certain cases arguments of elementFromPoint can be negative, so prevent setting targetElement to null
  555. 555  targetElement = document.elementFromPoint(touch.pageX - window.pageXOffset, touch.pageY - window.pageYOffset) || targetElement;
  556. 556  targetElement.fastClickScrollParent = this.targetElement.fastClickScrollParent;
  557. 557  }
  558. 558 
  559. 559  targetTagName = targetElement.tagName.toLowerCase();
  560. 560  if (targetTagName === 'label') {
  561. 561  forElement = this.findControl(targetElement);
  562. 562  if (forElement) {
  563. 563  this.focus(targetElement);
  564. 564  if (deviceIsAndroid) {
  565. 565  return false;
  566. 566  }
  567. 567 
  568. 568  targetElement = forElement;
  569. 569  }
  570. 570  } else if (this.needsFocus(targetElement)) {
  571. 571 
  572. 572  // Case 1: If the touch started a while ago (best guess is 100ms based on tests for issue #36) then focus will be triggered anyway. Return early and unset the target element reference so that the subsequent click will be allowed through.
  573. 573  // Case 2: Without this exception for input elements tapped when the document is contained in an iframe, then any inputted text won't be visible even though the value attribute is updated as the user types (issue #37).
  574. 574  if ((event.timeStamp - trackingClickStart) > 100 || (deviceIsIOS && window.top !== window && targetTagName === 'input')) {
  575. 575  this.targetElement = null;
  576. 576  return false;
  577. 577  }
  578. 578 
  579. 579  this.focus(targetElement);
  580. 580  this.sendClick(targetElement, event);
  581. 581 
  582. 582  // Select elements need the event to go through on iOS 4, otherwise the selector menu won't open.
  583. 583  // Also this breaks opening selects when VoiceOver is active on iOS6, iOS7 (and possibly others)
  584. 584  if (!deviceIsIOS || targetTagName !== 'select') {
  585. 585  this.targetElement = null;
  586. 586  event.preventDefault();
  587. 587  }
  588. 588 
  589. 589  return false;
  590. 590  }
  591. 591 
  592. 592  if (deviceIsIOS && !deviceIsIOS4) {
  593. 593 
  594. 594  // Don't send a synthetic click event if the target element is contained within a parent layer that was scrolled
  595. 595  // and this tap is being used to stop the scrolling (usually initiated by a fling - issue #42).
  596. 596  scrollParent = targetElement.fastClickScrollParent;
  597. 597  if (scrollParent && scrollParent.fastClickLastScrollTop !== scrollParent.scrollTop) {
  598. 598  return true;
  599. 599  }
  600. 600  }
  601. 601 
  602. 602  // Prevent the actual click from going though - unless the target node is marked as requiring
  603. 603  // real clicks or if it is in the whitelist in which case only non-programmatic clicks are permitted.
  604. 604  if (!this.needsClick(targetElement)) {
  605. 605  event.preventDefault();
  606. 606  this.sendClick(targetElement, event);
  607. 607  }
  608. 608 
  609. 609  return false;
  610. 610  };
  611. 611 
  612. 612 
  613. 613  /**
  614. 614   * On touch cancel, stop tracking the click.
  615. 615   *
  616. 616   * @returns {void}
  617. 617   */
  618. 618  FastClick.prototype.onTouchCancel = function() {
  619. 619  this.trackingClick = false;
  620. 620  this.targetElement = null;
  621. 621  };

  624. 624  /**
  625. 625   * Determine mouse events which should be permitted.

  627. 627   * @param {Event} event
  628. 628   * @returns {boolean}
  629. 629   */
  630. 630  FastClick.prototype.onMouse = function(event) {
  631. 631 
  632. 632  // If a target element was never set (because a touch event was never fired) allow the event
  633. 633  if (!this.targetElement) {
  634. 634  return true;
  635. 635  }
  636. 636 
  637. 637  if (event.forwardedTouchEvent) {
  638. 638  return true;
  639. 639  }
  640. 640 
  641. 641  // Programmatically generated events targeting a specific element should be permitted
  642. 642  if (!event.cancelable) {
  643. 643  return true;
  644. 644  }
  645. 645 
  646. 646  // Derive and check the target element to see whether the mouse event needs to be permitted;
  647. 647  // unless explicitly enabled, prevent non-touch click events from triggering actions,
  648. 648  // to prevent ghost/doubleclicks.
  649. 649  if (!this.needsClick(this.targetElement) || this.cancelNextClick) {
  650. 650 
  651. 651  // Prevent any user-added listeners declared on FastClick element from being fired.
  652. 652  if (event.stopImmediatePropagation) {
  653. 653  event.stopImmediatePropagation();
  654. 654  } else {
  655. 655 
  656. 656  // Part of the hack for browsers that don't support Event#stopImmediatePropagation (e.g. Android 2)
  657. 657  event.propagationStopped = true;
  658. 658  }
  659. 659 
  660. 660  // Cancel the event
  661. 661  event.stopPropagation();
  662. 662  event.preventDefault();
  663. 663 
  664. 664  return false;
  665. 665  }
  666. 666 
  667. 667  // If the mouse event is permitted, return true for the action to go through.
  668. 668  return true;
  669. 669  };

  672. 672  /**
  673. 673   * On actual clicks, determine whether this is a touch-generated click, a click action occurring
  674. 674   * naturally after a delay after a touch (which needs to be cancelled to avoid duplication), or
  675. 675   * an actual click which should be permitted.
  676. 676   *
  677. 677   * @param {Event} event
  678. 678   * @returns {boolean}
  679. 679   */
  680. 680  FastClick.prototype.onClick = function(event) {
  681. 681  var permitted;
  682. 682 
  683. 683  // It's possible for another FastClick-like library delivered with third-party code to fire a click event before FastClick does (issue #44). In that case, set the click-tracking flag back to false and return early. This will cause onTouchEnd to return early.
  684. 684  if (this.trackingClick) {
  685. 685  this.targetElement = null;
  686. 686  this.trackingClick = false;
  687. 687  return true;
  688. 688  }
  689. 689 
  690. 690  // Very odd behaviour on iOS (issue #18): if a submit element is present inside a form and the user hits enter in the iOS simulator or clicks the Go button on the pop-up OS keyboard the a kind of 'fake' click event will be triggered with the submit-type input element as the target.
  691. 691  if (event.target.type === 'submit' && event.detail === 0) {
  692. 692  return true;
  693. 693  }
  694. 694 
  695. 695  permitted = this.onMouse(event);
  696. 696 
  697. 697  // Only unset targetElement if the click is not permitted. This will ensure that the check for !targetElement in onMouse fails and the browser's click doesn't go through.
  698. 698  if (!permitted) {
  699. 699  this.targetElement = null;
  700. 700  }
  701. 701 
  702. 702  // If clicks are permitted, return true for the action to go through.
  703. 703  return permitted;
  704. 704  };
