  388.  388  Canvas.prototype.addText = function(layer, x, y, text, font, angle, width, halign, valign) {
  389.  389

  391.  391  positions = info.positions;
  392.  392

  394.  394

  396.  396  x -= info.width / 2;
  397.  397  } else if (halign == "right") {
  398.  398  x -= info.width;
  399.  399  }
  400.  400 
  401.  401  if (valign == "middle") {
  402.  402  y -= info.height / 2;
  403.  403  } else if (valign == "bottom") {
  404.  404  y -= info.height;
  405.  405  }
  406.  406 
  407.  407  // Determine whether this text already exists at this position.
  408.  408  // If so, mark it for inclusion in the next render pass.
  409.  409 
  410.  410  for (var i = 0, position; position = positions[i]; i++) {
  411.  411  if (position.x == x && position.y == y) {
  412.  412  position.active = true;
  413.  413  return;
  414.  414  }
  415.  415  }
  416.  416 
  417.  417  // If the text doesn't exist at this position, create a new entry
  418.  418

  420.  420  // while for subsequent ones we'll clone it.
  421.  421 
  422.  422  position = {
  423.  423  active: true,
  424.  424  rendered: false,
  425.  425  element: positions.length ? info.element.clone() : info.element,
  426.  426  x: x,
  427.  427  y: y
  428.  428  };
  429.  429 
  430.  430  positions.push(position);
  431.  431 
  432.  432  // Move the element to its final position within the container
  433.  433 
  434.  434  position.element.css({
  435.  435  top: Math.round(y),
  436.  436  left: Math.round(x),
  437.  437  'text-align': halign // In case the text wraps
  438.  438  });
  439.  439  };

  964.  964         function fillInSeriesOptions() {

  966.  966             var neededColors = series.length, maxIndex = -1, i;
  967.  967 
  968.  968             // Subtract the number of series that already have fixed colors or
  969.  969             // color indexes from the number that we still need to generate.
  970.  970 
  971.  971             for (i = 0; i < series.length; ++i) {
  972.  972                 var sc = series[i].color;
  973.  973                 if (sc != null) {
  974.  974                     neededColors--;
  975.  975                     if (typeof sc == "number" && sc > maxIndex) {
  976.  976                         maxIndex = sc;
  977.  977                     }
  978.  978                 }
  979.  979             }
  980.  980 
  981.  981             // If any of the series have fixed color indexes, then we need to
  982.  982             // generate at least as many colors as the highest index.
  983.  983 
  984.  984             if (neededColors <= maxIndex) {
  985.  985                 neededColors = maxIndex + 1;
  986.  986             }
  987.  987 
  988.  988             // Generate all the colors, using first the option colors and then
  989.  989             // variations on those colors once they're exhausted.
  990.  990 
  991.  991             var c, colors = [], colorPool = options.colors,
  992.  992                 colorPoolSize = colorPool.length, variation = 0;
  993.  993 
  994.  994             for (i = 0; i < neededColors; i++) {
  995.  995 
  996.  996                 c = $.color.parse(colorPool[i % colorPoolSize] || "#666");
  997.  997 
  998.  998                 // Each time we exhaust the colors in the pool we adjust
  999.  999                 // a scaling factor used to produce more variations on
  1000. 1000                 // those colors. The factor alternates negative/positive
  1001. 1001                 // to produce lighter/darker colors.
  1002. 1002 
  1003. 1003                 // Reset the variation after every few cycles, or else
  1004. 1004                 // it will end up producing only white or black colors.
  1005. 1005 
  1006. 1006                 if (i % colorPoolSize == 0 && i) {
  1007. 1007                     if (variation >= 0) {
  1008. 1008                         if (variation < 0.5) {
  1009. 1009                             variation = -variation - 0.2;
  1010. 1010                         } else variation = 0;
  1011. 1011                     } else variation = -variation;
  1012. 1012                 }
  1013. 1013 
  1014. 1014                 colors[i] = c.scale('rgb', 1 + variation);
  1015. 1015             }
  1016. 1016 
  1017. 1017             // Finalize the series options, filling in their colors
  1018. 1018 
  1019. 1019             var colori = 0, s;
  1020. 1020             for (i = 0; i < series.length; ++i) {
  1021. 1021                 s = series[i];
  1022. 1022 
  1023. 1023                 // assign colors
  1024. 1024                 if (s.color == null) {
  1025. 1025                     s.color = colors[colori].toString();
  1026. 1026                     ++colori;
  1027. 1027                 }
  1028. 1028                 else if (typeof s.color == "number")
  1029. 1029                     s.color = colors[s.color].toString();
  1030. 1030 
  1031. 1031                 // turn on lines automatically in case nothing is set
  1032. 1032                 if (s.lines.show == null) {
  1033. 1033                     var v, show = true;
  1034. 1034                     for (v in s)
  1035. 1035                         if (s[v] && s[v].show) {
  1036. 1036                             show = false;
  1037. 1037                             break;
  1038. 1038                         }
  1039. 1039                     if (show)
  1040. 1040                         s.lines.show = true;
  1041. 1041                 }
  1042. 1042 
  1043. 1043                 // If nothing was provided for lines.zero, default it to match
  1044. 1044                 // lines.fill, since areas by default should extend to zero.
  1045. 1045 
  1046. 1046                 if (s.lines.zero == null) {
  1047. 1047                     s.lines.zero = !!s.lines.fill;
  1048. 1048                 }
  1049. 1049 
  1050. 1050                 // setup axes
  1051. 1051                 s.xaxis = getOrCreateAxis(xaxes, axisNumber(s, "x"));
  1052. 1052                 s.yaxis = getOrCreateAxis(yaxes, axisNumber(s, "y"));
  1053. 1053             }
